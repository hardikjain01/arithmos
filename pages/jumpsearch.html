Jump Search

Let’s consider the following array: [ 0, 1, 2, 3, 4, 7, 9, 14, 15, 16, 17, 18, 19, 23, 24, 25, 26, 27, 28, 29 ]. Length of the array is 20. Jump search will find the value of 19 with the following steps assuming that the block size to be jumped is 4.

  1. Jump from index 0 to index 4;
  2. Jump from index 4 to index 8;
  3. Jump from index 12 to index 16;
  4. Since the element at index 16 is greater than 19 we will jump back a step to come to index 12.
  5. Perform linear search from index 12 to get the element 19.

package algorithms;

import java.util.Arrays;

/** A class to implement Jump Search
 * 
 * @author Hardik Jain
 *
 */
public class JumpSearch {
	private int[] data;
	private int size;
	
	public JumpSearch(int[] data) {
		this.data = data;
		this.size = this.data.length;
	}

	/**implements jump search
	 * 
	 * @param element to be searched
	 * @param size of the array
	 * @return index of element in data if found else -1
	 */
	public int search(int element) {
		
		// block size to be jumped
		int jump = (int)Math.floor(Math.sqrt(size));
		// Finding the block where element might be present 
        int low = 0;
        while (data[Math.min(jump, size)-1] < element) 
        { 
            low = jump; 
            jump += (int)Math.floor(Math.sqrt(size)); 
            if (low >= size) 
                return -1; 
        }
        
        /*
         * implementing linear search on the block 
         * starting from low
         */
        while (data[low] < element) 
        { 
            low++; 
            /*
             * If we reached next block or end of array, 
             * element is not present. 
             */
            if (low == Math.min(jump, size)) 
                return -1; 
        } 
  
        // If element is found 
        if (data[low] == element) 
            return low;
        
		// If element is not found in the array
		return -1;
	}
	
	/**
	 * @return the size
	 */
	public int getSize() {
		return size;
	}
	
	/**
	 *  to print the array
	 */
	public void print() {
		System.out.print("[ ");
		for(int i=0; i< size; i++) {
			System.out.print(data[i]);
			if(i<size-1) {
				System.out.print(", ");
			}
		}
		System.out.print(" ]");
	}
	
	public static void main(String[] args) {
		int array[] = { 24, 18, 4, 19, 17, 9, 23, 0, 2, 3, 29, 14, 16, 7, 15, 25, 27, 28, 26, 1 };
		Arrays.sort(array);
		JumpSearch JS = new JumpSearch(array);
		JS.print();
		int toFind = 19;
		int index = JS.search(toFind);
		
		if(index == -1) {
			System.out.print("\nElement not found");
		}
		else {
			System.out.print("\n" + toFind + " found at index " + index);
		}
	}
}

                                    
                                    
Output:

                                    
[ 0, 1, 2, 3, 4, 7, 9, 14, 15, 16, 17, 18, 19, 23, 24, 25, 26, 27, 28, 29 ]
19 found at index 12

The time complexity of above algorithm is O(√n).
