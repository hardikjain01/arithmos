Binary Search

We basically ignore half of the elements just after one comparison.

  1. Compare x with the middle element.
  2. If x matches with middle element, we return the mid index.
  3. Else If x is greater than the mid element, then x can only lie in right half subarray after the mid element. So we repeat for right half.
  4. Else (x is smaller) repeat for the left half.
back 
                                    
package algorithms;

import java.util.Arrays;

/** A class to implement Binary Search
 * 
 * @author Hardik Jain
 *
 */

public class BinarySearch {
	private int[] data;
	private int size;
	
	public BinarySearch(int[] data) {
		this.data = data;
		this.size = this.data.length;
	}
	
	/**implements binary search
	 * 
	 * @param element to be searched
	 * @param size of the array
	 * @return index of element in data if found else -1
	 */
	public int search(int element) {
		// For binary search array should be sorted
		int low = 0;
		int high = size-1;
		int mid;
		while(low<=high) {
			mid = (low + high) / 2;
			// element found at the middle of the range
			if(data[mid] == element) {
				return mid;
			}
			// element must be in lower half of the range
			else if(data[mid] > element) {
				high = mid-1;
			}
			// element must be in upper half of the range
			else {
				low = mid+1;
			}
		}
		// If element is not found in the array
		return -1;
	}

	/**
	 * @return the size
	 */
	public int getSize() {
		return size;
	}
	
	/**
	 *  to print the array
	 */
	public void print() {
		System.out.print("[ ");
		for(int i=0; i< size; i++) {
			System.out.print(data[i]);
			if(i<size-1) {
				System.out.print(", ");
			}
		}
		System.out.print(" ]");
	}
	
	public static void main(String[] args) {
		int array[] = { 24, 18, 4, 19, 17, 9, 23, 0, 2, 3, 29, 14, 16, 7, 15, 25, 27, 28, 26, 1 };
		Arrays.sort(array);
		BinarySearch BS = new BinarySearch(array);
		BS.print();
		int toFind = 14;
		int index = BS.search(toFind);
		
		if(index == -1) {
			System.out.print("\nElement not found");
		}
		else {
			System.out.print("\n" + toFind + " found at index " + index);
		}
	}
	
}

                                    
                                    
Output:

                                    
[ 0, 1, 2, 3, 4, 7, 9, 14, 15, 16, 17, 18, 19, 23, 24, 25, 26, 27, 28, 29 ]
14 found at index 7

The time complexity of above algorithm is O(log(n)).
